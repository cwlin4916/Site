\noindent \cite[7]{Lod92}, \cite[B]{NS18}.
\begin{nul} Let us define a few notions.
\begin{itemize}
\item $\PoSet$ be the cat. of posets with order preserving maps.
\item $\PoSet^{B \B Z}$ be functor category.
\begin{itemize}
\item Objects are posets with $\B Z$-action.
\item Moprh. consists of comm. dia.
\[
\begin{tikzcd}
A \ar[d,"n"] \ar[r] & B \ar[d,"n"] \\
A \ar[r] & B
\end{tikzcd}
\]
\noindent for all $n \in \B Z$.
\end{itemize}
\end{itemize}
\item $\Lambda_\infty \subset \PoSet^{\B Z}$ full subcat.
\end{nul}
